376     @Override
377     public void execute() throws MojoExecutionException, MojoFailureException {
378         // Check if plugin run should be skipped
379         if (this.skip) {
380             getLog().info("Makeself is skipped");
381             return;
382         }
383 
384         // Setup make self files
385         this.extractMakeself();
386 
387         // Get OS Name
388         boolean isWindows = System.getProperty("os.name").startsWith("Windows");
389 
390         try {
391             // Output version of bash
392             getLog().debug("Execute Bash Version");
393             execute(Arrays.asList("bash", "--version"), !ATTACH_ARTIFACT);
394 
395             // Output version of makeself.sh
396             getLog().debug("Execute Makeself Version");
397             execute(Arrays.asList("bash", makeself.getAbsolutePath(), "--version"), !ATTACH_ARTIFACT);
398 
399             // If help arguments supplied, write output and get out of code.
400             String helpArgs = helpArgs();
401             if (!helpArgs.isEmpty()) {
402                 getLog().debug("Execute Makeself Help");
403                 execute(Arrays.asList("bash", makeself.getAbsolutePath(), helpArgs), !ATTACH_ARTIFACT);
404                 return;
405             }
406 
407             // Basic Configuration
408             getLog().debug("Loading Makeself Basic Configuration");
409             List<String> target = new ArrayList<>();
410             target.addAll(Arrays.asList("bash", makeself.getAbsolutePath()));
411             target.addAll(loadArgs());
412             target.add(buildTarget.concat(archiveDir));
413             target.add(buildTarget.concat(fileName));
414             target.add(label);
415             target.add(startupScript);
416             if (scriptArgs != null) {
417                 target.addAll(Arrays.asList(scriptArgs.split("\\s+")));
418             }
419 
420             // Indicate makeself running
421             getLog().info("Running makeself build");
422 
423             // Execute main run of makeself.sh
424             getLog().debug("Execute Makeself Build");
425             execute(target, ATTACH_ARTIFACT);
426 
427             // Output info on file makeself created
428             getLog().debug("Execute Makeself Info on Resulting Shell Script");
429             execute(Arrays.asList("bash", buildTarget.concat(fileName), "--info"), !ATTACH_ARTIFACT);
430 
431             // Output list on file makeself created (non windows need)
432             if (!isWindows) {
433                 getLog().debug("Execute Makeself List on Resulting Shell Script");
434                 execute(Arrays.asList("bash", buildTarget.concat(fileName), "--list"), !ATTACH_ARTIFACT);
435             }
436 
437             // auto run script
438             if (this.autoRun) {
439                 getLog().info("Auto-run created shell (this may take a few minutes)");
440                 execute(Arrays.asList("bash", buildTarget.concat(fileName)), !ATTACH_ARTIFACT);
441             }
442         } catch (IOException e) {
443             getLog().error("", e);
444             if (e.getMessage().contains("Cannot run program \"bash\"")) {
445                 // Note: we printed the full stack trace already so don't do it again.
446                 if (isWindows) {
447                     throw new MojoFailureException(
448                             "Configure Bash with Cygwin or Git for Windows by adding '/usr/bin' to environment 'Path' variable to execute this plugin");
449                 }
450                 throw new MojoFailureException("Configure Bash to execute this plugin");
451             }
452         } catch (InterruptedException e) {
453             getLog().error("", e);
454             // restore interruption status of the corresponding thread
455             Thread.currentThread().interrupt();
456         }
457     }

459     private void execute(List<String> target, boolean attach) throws IOException, InterruptedException {
460         // Create Process Builder
461         ProcessBuilder processBuilder = new ProcessBuilder(target);
462         processBuilder.redirectErrorStream(true);
463 
464         // Create Process
465         Process process = processBuilder.start();
466 
467         // Write process output
468         try (BufferedReader reader = new BufferedReader(
469                 new InputStreamReader(process.getInputStream(), StandardCharsets.UTF_8))) {
470             String line = "";
471             while ((line = reader.readLine()) != null) {
472                 getLog().info(line);
473             }
474             getLog().info("");
475         }
476 
477         // Wait for process completion
478         int status = process.waitFor();
479         if (status > 0) {
480             getLog().error(Joiner.on(" ").join("makeself failed with error status:", status));
481         }
482 
483         // Attach artifact to maven build for install/deploy/release on success
484         if (status == 0 && attach) {
485             projectHelper.attachArtifact(project, "sh", new File(buildTarget.concat(fileName)));
486         }
487     }

489     /**
490      * Extract makeself.
491      */
492     private void extractMakeself() {
493         getLog().debug("Extracting Makeself");
494 
495         // Create makeself directory
496         File makeselfTemp = new File(targetDirectory.getAbsolutePath());
497         if (!makeselfTemp.exists() && !makeselfTemp.mkdir()) {
498             getLog().error(Joiner.on(" ").join("Unable to make directory", targetDirectory.getAbsolutePath()));
499             return;
500         } else {
501             getLog().debug(Joiner.on(" ").join("Created directory for", targetDirectory.getAbsolutePath()));
502         }
503 
504         ClassLoader classloader = this.getClass().getClassLoader();
505 
506         // Write makeself script
507         makeself = new File(Joiner.on("").join(targetDirectory, "/makeself.sh"));
508         if (!makeself.exists()) {
509             getLog().debug("Writing makeself.sh");
510             try (InputStream link = classloader.getResourceAsStream("META-INF/makeself/makeself.sh")) {
511                 Path path = makeself.getAbsoluteFile().toPath();
512                 Files.copy(link, path);
513                 setFilePermissions(makeself);
514                 setPosixFilePermissions(path);
515             } catch (IOException e) {
516                 getLog().error("", e);
517             }
518         }
519 
520         // Write makeself-header script
521         File makeselfHeader = new File(Joiner.on("").join(targetDirectory, "/makeself-header.sh"));
522         if (!makeselfHeader.exists()) {
523             getLog().debug("Writing makeself-header.sh");
524             try (InputStream link = classloader.getResourceAsStream("META-INF/makeself/makeself-header.sh")) {
525                 Path path = makeselfHeader.getAbsoluteFile().toPath();
526                 Files.copy(link, path);
527                 setFilePermissions(makeselfHeader);
528                 setPosixFilePermissions(path);
529             } catch (IOException e) {
530                 getLog().error("", e);
531             }
532         }
533     }
534 
535     private void setFilePermissions(File file) {
536         if (!file.setExecutable(true, true)) {
537             getLog().error(Joiner.on(" ").join("Unable to set executable:", file.getName()));
538         } else {
539             getLog().debug(Joiner.on(" ").join("Set executable for", file.getName()));
540         }
541     }
542 
543     private void setPosixFilePermissions(Path path) {
544         final Set<PosixFilePermission> permissions = PosixFilePermissions.fromString("rwxr-xr--");
545 
546         try {
547             Files.setPosixFilePermissions(path, permissions);
548             getLog().debug(Joiner.on(" ").join("Set Posix File Permissions for", path, "as", permissions));
549         } catch (IOException e) {
550             getLog().error("Failed attempted Posix permissions", e);
551         } catch (UnsupportedOperationException e) {
552             // Attempting but don't care about status if it fails
553             getLog().debug("Failed attempted Posix permissions", e);
554         }
555     }

656 
657         // --ssl-pass-src src : Use the given src as the source of password to encrypt the data using OpenSSL. See
658         // \"PASS PHRASE ARGUMENTS\" in man openssl. If this option is not supplied, the user wil be asked to enter
659         // encryption pasword on the current terminal.
660         if (sslPasswd != null) {
661             args.add("--ssl-pass-src");
662             args.add(sslPassSrc);
663         }
664 
665         // --ssl-no-md : Do not use \"-md\" option not supported by older OpenSSL.
666         if (isTrue(sslNoMd)) {
667             args.add("--ssl-no-md");
668         }
669 
670         // --compress : Use the UNIX compress command to compress the data. This should be the default on all platforms
671         // that don't have gzip available.
672         if (isTrue(compress)) {
673             args.add("--compress");
674         }
675 
676         // --nocomp : Do not use any compression for the archive, which will then be an uncompressed TAR.
677         if (isTrue(nocomp)) {
678             args.add("--nocomp");
679         }
680 
681         // --complevel : Specify the compression level for gzip, bzip2, pbzip2, xz, lzo or lz4. (defaults to 9)
682         if (complevel != null) {
683             args.add("--complevel");
684             args.add(complevel.toString());
685         }
686 
687         // --notemp : The generated archive will not extract the files to a temporary directory, but in a new directory
688         // created in the current directory. This is better to distribute software packages that may extract and compile
689         // by themselves (i.e. launch the compilation through the embedded script).
690         if (isTrue(notemp)) {
691             args.add("--notemp");
692         }
693 
694         // --current : Files will be extracted to the current directory, instead of in a sub-directory. This option
695         // implies --notemp above.
696         if (isTrue(current)) {
697             args.add("--current");
698         }
699 
700         // --follow : Follow the symbolic links inside of the archive directory, i.e. store the files that are being
701         // pointed to instead of the links themselves.
702         if (isTrue(follow)) {
703             args.add("--follow");
704         }
705 
706         // --append (new in 2.1.x): Append data to an existing archive, instead of creating a new one. In this mode, the
707         // settings from the original archive are reused (compression type, label, embedded script), and thus don't need
708         // to be specified again on the command line.
709         if (isTrue(append)) {
710             args.add("--append");
711         }
712 
713         // --header : Makeself 2.0 uses a separate file to store the header stub, called makeself-header.sh. By default,
714         // it is assumed that it is stored in the same location as makeself.sh. This option can be used to specify its
715         // actual location if it is stored someplace else.
716         if (headerFile != null) {
717             args.add("--header");
718             args.add(headerFile.toString());
719         }
720 
721         // --copy : Upon extraction, the archive will first extract itself to a temporary directory. The main
722         // application of this is to allow self-contained installers stored in a Makeself archive on a CD, when the
723         // installer program will later need to unmount the CD and allow a new one to be inserted. This prevents
724         // "File system busy" errors for installers that span multiple CDs.
725         if (isTrue(copy)) {
726             args.add("--copy");
727         }
728 
729         // --nox11 : Disable the automatic spawning of a new terminal in X11.
730         if (isTrue(nox11)) {
731             args.add("--nox11");
732         }
733 
734         // --nowait : When executed from a new X11 terminal, disable the user prompt at the end of the script execution.
735         if (isTrue(nowait)) {
736             args.add("--nowait");
737         }
738 
739         // --nomd5 : Disable the creation of a MD5 checksum for the archive. This speeds up the extraction process if
740         // integrity checking is not necessary.
741         if (isTrue(nomd5)) {
742             args.add("--nomd5");
743         }
744 
745         // --nocrc : Disable the creation of a CRC checksum for the archive. This speeds up the extraction process if
746         // integrity checking is not necessary.
747         if (isTrue(nocrc)) {
748             args.add("--nocrc");
749         }
750 
751         // --sha256 : Compute a SHA256 checksum for the archive.
752         if (isTrue(sha256)) {
753             args.add("--sha256");
754         }
755 
756         // --lsm file : Provide and LSM file to makeself, that will be embedded in the generated archive. LSM files are
757         // describing a software package in a way that is easily parseable. The LSM entry can then be later retrieved
758         // using the --lsm argument to the archive. An example of a LSM file is provided
759         // with Makeself.
760         if (lsmFile != null) {
761             args.add("--lsm");
762             args.add(lsmFile);
763         }
764 
765         // --gpg-extra opt : Append more options to the gpg command line.
766         if (gpgExtraOpt != null) {
767             args.add("--gpg-extra");
768             args.add(gpgExtraOpt);
769         }
770 
771         // --tar-extra opt : Append more options to the tar command line.
772         // For instance, in order to exclude the .git directory from the packaged archive directory using the GNU tar,
773         // one can use makeself.sh --tar-extra "--exclude=.git" ...
774         if (tarExtraOpt != null) {
775             args.add("--tar-extra");
776             args.add(tarExtraOpt);
777         }
778 
779         // --untar-extra opt : Append more options to the during the extraction of the tar archive.
780         if (untarExtraOpt != null) {
781             args.add("--untar-extra");
782             args.add(untarExtraOpt);
783         }
784 
785         // --keep-umask : Keep the umask set to shell default, rather than overriding when executing self-extracting
786         // archive.
787         if (isTrue(keepUmask)) {
788             args.add("--keep-umask");
789         }
790 
791         // --export-conf : Export configuration variables to startup_script"
792         if (isTrue(exportConf)) {
793             args.add("--export-conf");
794         }
795 
796         // --packaging-date date : Use provided string as the packaging date instead of the current date.
797         if (packagingDate != null) {
798             args.add("--packaging-date");
799             args.add(packagingDate);
800         }
801 
802         // --license : Append a license file.
803         if (licenseFile != null) {
804             args.add("--license");
805             args.add(licenseFile);
806         }
807 
808         // --nooverwrite : Do not extract the archive if the specified target directory already exists.
809         if (isTrue(nooverwrite)) {
810             args.add("--nooverwrite");
811         }
812 
813         // --help-header file : Add a header to the archive's --help output.
814         if (helpHeaderFile != null) {
815             args.add("--helpHeaderFile");
816             args.add(helpHeaderFile);
817         }
818 
819         return args;
820     }
821 
822     private boolean isTrue(Boolean value) {
823         if (value != null) {
824             return value.booleanValue();
825         }
826         return false;
827     }
