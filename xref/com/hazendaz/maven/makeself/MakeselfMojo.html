499     @Override
500     public void execute() throws MojoExecutionException, MojoFailureException {
501         // Check if plugin run should be skipped
502         if (this.skip) {
503             getLog().info("Makeself is skipped");
504             return;
505         }
506 
507         // Validate archive directory exists
508         File file = new File(buildTarget.concat(archiveDir));
509         if (!file.exists()) {
510             throw new MojoExecutionException("ArchiveDir: missing '" + buildTarget.concat(archiveDir) + "'");
511         }
512 
513         // Validate inline script or startup script file
514         if (inlineScript) {
515             // Validate inline script has script args
516             if (scriptArgs == null) {
517                 throw new MojoExecutionException("ScriptArgs required when running inlineScript");
518             }
519         } else {
520             // Validate startupScript file starts with './'
521             if (!startupScript.startsWith("./")) {
522                 throw new MojoExecutionException("StartupScript required to start with './'");
523             }
524 
525             // Validate startupScript file exists
526             file = new File(buildTarget.concat(archiveDir).concat(startupScript.substring(1)));
527             if (!file.exists()) {
528                 throw new MojoExecutionException("StartupScript: missing '"
529                         + buildTarget.concat(archiveDir).concat(startupScript.substring(1)) + "'");
530             }
531         }
532 
533         // Setup make self files
534         this.extractMakeself();
535 
536         // Check git setup
537         if (MakeselfMojo.WINDOWS) {
538             this.checkGitSetup();
539         }
540 
541         try {
542             // Output version of bash
543             getLog().debug("Execute Bash Version");
544             execute(Arrays.asList(gitPath + "bash", "--version"), !ATTACH_ARTIFACT);
545 
546             // Output version of makeself.sh
547             getLog().debug("Execute Makeself Version");
548             execute(Arrays.asList(gitPath + "bash", makeself.getAbsolutePath(), "--version"), !ATTACH_ARTIFACT);
549 
550             // If help arguments supplied, write output and get out of code.
551             String helpArgs = helpArgs();
552             if (!helpArgs.isEmpty()) {
553                 getLog().debug("Execute Makeself Help");
554                 execute(Arrays.asList(gitPath + "bash", makeself.getAbsolutePath(), helpArgs), !ATTACH_ARTIFACT);
555                 return;
556             }
557 
558             // Basic Configuration
559             getLog().debug("Loading Makeself Basic Configuration");
560             List<String> target = new ArrayList<>();
561             target.addAll(Arrays.asList(gitPath + "bash", makeself.getAbsolutePath()));
562             target.addAll(loadArgs());
563             target.add(buildTarget.concat(archiveDir));
564             target.add(buildTarget.concat(fileName));
565             target.add(label);
566             target.add(startupScript);
567             if (scriptArgs != null) {
568                 target.addAll(scriptArgs);
569             }
570             if (cleanupScript != null) {
571                 target.add(cleanupScript);
572                 // Pass original arguments again
573                 if (scriptArgs != null) {
574                     target.addAll(scriptArgs);
575                 }
576                 // Pass cleanup arguments
577                 if (cleanupArgs != null) {
578                     target.addAll(cleanupArgs);
579                 }
580             }
581 
582             // Indicate makeself running
583             getLog().info("Running makeself build");
584 
585             // Execute main run of makeself.sh
586             getLog().debug("Execute Makeself Build");
587             execute(target, ATTACH_ARTIFACT);
588 
589             // Output info on file makeself created
590             getLog().debug("Execute Makeself Info on Resulting Shell Script");
591             execute(Arrays.asList(gitPath + "bash", buildTarget.concat(fileName), "--info"), !ATTACH_ARTIFACT);
592 
593             // Output list on file makeself created (non windows need)
594             if (!MakeselfMojo.WINDOWS) {
595                 getLog().debug("Execute Makeself List on Resulting Shell Script");
596                 execute(Arrays.asList(gitPath + "bash", buildTarget.concat(fileName), "--list"), !ATTACH_ARTIFACT);
597             }
598 
599             // auto run script
600             if (this.autoRun) {
601                 getLog().info("Auto-run created shell (this may take a few minutes)");
602                 execute(Arrays.asList(gitPath + "bash", buildTarget.concat(fileName)), !ATTACH_ARTIFACT);
603             }
604         } catch (IOException e) {
605             getLog().error("", e);
606         } catch (InterruptedException e) {
607             getLog().error("", e);
608             // restore interruption status of the corresponding thread
609             Thread.currentThread().interrupt();
610         }
611     }

613     private void execute(List<String> target, boolean attach) throws IOException, InterruptedException {
614 
615         // Log execution target
616         getLog().debug("Execution commands: " + target);
617 
618         // Create Process Builder
619         ProcessBuilder processBuilder = new ProcessBuilder(target);
620         processBuilder.redirectErrorStream(true);
621 
622         // Add portable git to windows environment
623         if (MakeselfMojo.WINDOWS) {
624             Map<String, String> envs = processBuilder.environment();
625             getLog().debug("Environment Variables: " + envs);
626             final String location = localRepository.getBasedir() + File.separator + this.portableGit.getName()
627                     + File.separator + this.portableGit.getVersion();
628             // Windows cmd/powershell shows "Path" in this case
629             if (envs.get("Path") != null) {
630                 envs.put("Path", location + "/usr/bin;" + envs.get("Path"));
631                 getLog().debug("Environment Path Variable: " + envs.get("Path"));
632                 // Windows bash shows "PATH" in this case and has issues with spacing as in 'Program Files'
633             } else if (envs.get("PATH") != null) {
634                 envs.put("PATH",
635                         location + "/usr/bin;" + envs.get("PATH").replace("Program Files", "\"Program Files\""));
636                 getLog().debug("Environment Path Variable: " + envs.get("PATH"));
637             }
638         }
639 
640         // Create Process
641         Process process = processBuilder.start();
642 
643         // Write process output
644         try (BufferedReader reader = new BufferedReader(
645                 new InputStreamReader(process.getInputStream(), StandardCharsets.UTF_8))) {
646             String line = "";
647             while ((line = reader.readLine()) != null) {
648                 getLog().info(line);
649             }
650             getLog().info("");
651         }
652 
653         // Wait for process completion
654         int status = process.waitFor();
655         if (status > 0) {
656             getLog().error(Joiner.on(" ").join("makeself failed with error status:", status));
657         }
658 
659         // Attach artifact to maven build for install/deploy/release on success
660         if (status == 0 && attach) {
661             projectHelper.attachArtifact(project, this.extension, this.classifier,
662                     new File(buildTarget, FilenameUtils.getName(fileName)));
663         }
664     }

669     private void extractMakeself() {
670         getLog().debug("Extracting Makeself");
671 
672         // Create makeself directory
673         File makeselfTemp = new File(targetDirectory.getAbsolutePath());
674         if (!makeselfTemp.exists() && !makeselfTemp.mkdirs()) {
675             getLog().error(Joiner.on(" ").join("Unable to make directory", targetDirectory.getAbsolutePath()));
676             return;
677         } else {
678             getLog().debug(Joiner.on(" ").join("Created directory for", targetDirectory.getAbsolutePath()));
679         }
680 
681         ClassLoader classloader = this.getClass().getClassLoader();
682 
683         // Write makeself script
684         makeself = new File(targetDirectory, "makeself.sh");
685         if (!makeself.exists()) {
686             getLog().debug("Writing makeself.sh");
687             try (InputStream link = classloader.getResourceAsStream("META-INF/makeself/makeself.sh")) {
688                 Path path = makeself.getAbsoluteFile().toPath();
689                 Files.copy(link, path);
690                 setFilePermissions(makeself);
691                 setPosixFilePermissions(path);
692             } catch (IOException e) {
693                 getLog().error("", e);
694             }
695         }
696 
697         // Write makeself-header script
698         File makeselfHeader = new File(targetDirectory, "makeself-header.sh");
699         if (!makeselfHeader.exists()) {
700             getLog().debug("Writing makeself-header.sh");
701             try (InputStream link = classloader.getResourceAsStream("META-INF/makeself/makeself-header.sh")) {
702                 Path path = makeselfHeader.getAbsoluteFile().toPath();
703                 Files.copy(link, path);
704                 setFilePermissions(makeselfHeader);
705                 setPosixFilePermissions(path);
706             } catch (IOException e) {
707                 getLog().error("", e);
708             }
709         }
710     }

726     /**
727      * Extract Portable Git.
728      */
729     private void extractPortableGit() {
730         final String location = localRepository.getBasedir() + File.separator + this.portableGit.getName()
731                 + File.separator + this.portableGit.getVersion();
732         if (new File(location).exists()) {
733             getLog().debug("Existing 'PortableGit' folder found at " + location);
734             gitPath = location + "/usr/bin/";
735             return;
736         }
737 
738         getLog().info("Loading portable git");
739         final Artifact artifact = repositorySystem.createArtifactWithClassifier(this.portableGit.getGroupId(),
740                 this.portableGit.getArtifactId(), this.portableGit.getVersion(), this.portableGit.getType(),
741                 this.portableGit.getClassifier());
742 
743         final ArtifactResolutionRequest artifactResolutionRequest = new ArtifactResolutionRequest();
744         artifactResolutionRequest.setArtifact(artifact);
745         artifactResolutionRequest.setResolveTransitively(true);
746         artifactResolutionRequest.setLocalRepository(localRepository);
747         artifactResolutionRequest.setRemoteRepositories(remoteRepositories);
748         repositorySystem.resolve(artifactResolutionRequest);
749 
750         this.installGit(artifact, location);
751     }

753     /**
754      * Install Git extracts git to .m2/repository under PortableGit.
755      *
756      * @param artifact
757      *            the maven artifact representation for git
758      * @param location
759      *            the location in maven repository to store portable git
760      */
761     private void installGit(final Artifact artifact, final String location) {
762         File currentFile = null;
763 
764         // Unzip 'tar.gz' from repository under 'com/github/hazendaz/git/git-for-windows' into
765         // .m2/repository/PortableGit
766         try (TarArchiveInputStream tarArchiveInputStream = new TarArchiveInputStream(new GzipCompressorInputStream(
767                 new BufferedInputStream(Files.newInputStream(artifact.getFile().toPath()))))) {
768             TarArchiveEntry entry;
769             while ((entry = tarArchiveInputStream.getNextTarEntry()) != null) {
770                 if (entry.isDirectory()) {
771                     continue;
772                 }
773                 currentFile = new File(localRepository.getBasedir() + File.separator + this.portableGit.getName(),
774                         entry.getName());
775                 File parent = currentFile.getParentFile();
776                 if (!parent.exists()) {
777                     parent.mkdirs();
778                 }
779                 getLog().debug("Current file: " + currentFile.getName());
780                 Files.copy(tarArchiveInputStream, currentFile.toPath(), StandardCopyOption.REPLACE_EXISTING);
781             }
782         } catch (IOException e) {
783             getLog().error("", e);
784         }
785 
786         try {
787             if (currentFile != null) {
788                 // Extract Portable Git
789                 getLog().debug("Extract Portable Git");
790                 execute(Arrays.asList(currentFile.toPath().toString(), "-y", "-o", location), !ATTACH_ARTIFACT);
791                 gitPath = location + "/usr/bin/";
792             }
793         } catch (IOException e) {
794             getLog().error("", e);
795         } catch (InterruptedException e) {
796             getLog().error("", e);
797             // restore interruption status of the corresponding thread
798             Thread.currentThread().interrupt();
799         }
800     }

804             getLog().error(Joiner.on(" ").join("Unable to set executable:", file.getName()));

807         }

810     private void setPosixFilePermissions(Path path) {
811         final Set<PosixFilePermission> permissions = PosixFilePermissions.fromString("rwxr-xr--");
812 
813         try {
814             Files.setPosixFilePermissions(path, permissions);
815             getLog().debug(Joiner.on(" ").join("Set Posix File Permissions for", path, "as", permissions));
816         } catch (IOException e) {
817             getLog().error("Failed attempted Posix permissions", e);
818         } catch (UnsupportedOperationException e) {
819             // Attempting but don't care about status if it fails
820             getLog().debug("Failed attempted Posix permissions", e);
821         }
822     }

843      *
844      * @return the string
845      */
846     private List<String> loadArgs() {
847         getLog().debug("Loading arguments");
848 
849         List<String> args = new ArrayList<>(50);
850 
851         // --gzip : Use gzip for compression (the default on platforms on which gzip is commonly available, like Linux)
852         if (isTrue(gzip)) {
853             args.add("--gzip");
854         }
855 
856         // --bzip2 : Use bzip2 instead of gzip for better compression. The bzip2 command must be available in the
857         // command path. It is recommended that the archive prefix be set to something like '.bz2.run', so that
858         // potential users know that they'll need bzip2 to extract it.
859         if (isTrue(bzip2)) {
860             args.add("--bzip2");
861         }
862 
863         // --pbzip2 : Use pbzip2 instead of gzip for better and faster compression on machines having multiple CPUs.
864         // The pbzip2 command must be available in the command path. It is recommended that the archive prefix be
865         // set to something like '.bz2.run', so that potential users know that they'll need bzip2 to extract it.
866         if (isTrue(pbzip2)) {
867             args.add("--pbzip2");
868         }
869 
870         // --xz : Use xz instead of gzip for better compression. The xz command must be available in the command path.
871         // It is recommended that the archive prefix be set to something like '.xz.run' for the archive, so that
872         // potential users know that they'll need xz to extract it.
873         if (isTrue(xz)) {
874             args.add("--xz");
875         }
876 
877         // --lzo : Use lzop instead of gzip for better compression. The lzop command must be available in the command
878         // path. It is recommended that the archive prefix be set to something like '.lzo.run' for the archive, so
879         // that potential users know that they'll need lzop to extract it.
880         if (isTrue(lzo)) {
881             args.add("--lzo");
882         }
883 
884         // --lz4 : Use lz4 instead of gzip for better compression. The lz4 command must be available in the command
885         // path. It is recommended that the archive prefix be set to something like '.lz4.run' for the archive, so
886         // that potential users know that they'll need lz4 to extract it.
887         if (isTrue(lz4)) {
888             args.add("--lz4");
889         }
890 
891         // --pigz : Use pigz for compression.
892         if (isTrue(pigz)) {
893             args.add("--pigz");
894         }
895 
896         // --zstd : Use zstd for compression.
897         if (isTrue(zstd)) {
898             args.add("--zstd");
899         }
900 
901         // --base64 : Encode the archive to ASCII in Base64 format (base64 command required).
902         if (isTrue(base64)) {
903             args.add("--base64");
904         }
905 
906         // --gpg-encrypt : Encrypt the archive using gpg -ac -z $COMPRESS_LEVEL. This will prompt for a password to
907         // encrypt with. Assumes that potential users have gpg installed.
908         if (isTrue(gpgEncrypt)) {
909             args.add("--gpg-encrypt");
910         }
911 
912         // --gpg-asymmetric-encrypt-sign : Instead of compressing, asymmetrically encrypt and sign the data using GPG
913         if (isTrue(gpgAsymmetricEncryptSign)) {
914             args.add("--gpg-asymmetric-encrypt-sign");
915         }
916 
917         // --ssl-encrypt : Encrypt the archive using openssl aes-256-cbc -a -salt. This will prompt for a password to
918         // encrypt with. Assumes that the potential users have the OpenSSL tools installed.
919         if (isTrue(sslEncrypt)) {
920             args.add("--ssl-encrypt");
921         }
922 
923         // --ssl-passwd pass : Use the given password to encrypt the data using OpenSSL.
924         if (sslPasswd != null) {
925             args.add("--ssl-passwd");
926             args.add(sslPasswd);
927         }
928 
929         // --ssl-pass-src src : Use the given src as the source of password to encrypt the data using OpenSSL. See
930         // \"PASS PHRASE ARGUMENTS\" in man openssl. If this option is not supplied, the user wil be asked to enter
931         // encryption pasword on the current terminal.
932         if (sslPasswd != null) {
933             args.add("--ssl-pass-src");
934             args.add(sslPassSrc);
935         }
936 
937         // --ssl-no-md : Do not use \"-md\" option not supported by older OpenSSL.
938         if (isTrue(sslNoMd)) {
939             args.add("--ssl-no-md");
940         }
941 
942         // --compress : Use the UNIX compress command to compress the data. This should be the default on all platforms
943         // that don't have gzip available.

945             args.add("--compress");

954         if (complevel != null) {
955             args.add("--complevel");
956             args.add(complevel.toString());
957         }
958 
959         // --threads thds : Number of threads to be used by compressors that support parallelization.
960         // Omit to use compressor's default. Most useful (and required) for opting into xz's threading,
961         // usually with '--threads=0' for all available cores.pbzip2 and pigz are parallel by default,
